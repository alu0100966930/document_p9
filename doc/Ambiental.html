    # File lib/p6/alimentos.rb, line 279
def get_gas
        co = 0
        i = 0
        j = 0
        @listaGr.each do |gramos|
                gr = gramos
                i = i + 1
                @listaAl.each do |element|
                        j = j + 1
                        if (i == j)
                                co = co + gr*element.gei/100
                        end
                end 
                j = 0
        end
        @gas = co
        return co 
end

def get_terreno
        t = 0
        i = 0
        j = 0
        @listaGr.each do |gramos|
                gr = gramos
                i = i + 1
                @listaAl.each do |element|
                        j = j + 1
                        if (i == j)
                                t = t + gr*element.terreno/100
                        end
                end
                j = 0
        end
        @ter = t
        return t
end
